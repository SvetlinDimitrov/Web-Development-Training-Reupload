01. Employee Address
select e.employee_id,
e.job_title,
e.address_id,
(select address_text from addresses where address_id = e.address_id)
from employees as e
order by address_id
limit 5;

02. Addresses with Towns
select first_name,last_name,
(select name from towns as t where t.town_id  = 
	(
		select town_id 
		from addresses as a 
		where a.address_id = e.address_id
	)
) as town ,
(
	select address_text 
    from addresses 
    where address_id = e.address_id
) as address_text
from employees as e
order by first_name , last_name
limit 5;

03. Sales Employee
select employee_id ,
first_name,
last_name,
(
	select name 
    from departments 
    where department_id = e.department_id
) as department_name
from employees as e
having department_name = 'Sales'
order by employee_id desc;

04. Employee Departments
select employee_id ,
first_name,
salary,
(
	select name 
    from departments 
    where department_id = e.department_id
) as department_name
from employees as e
having salary > 15000
order by department_id desc
limit 5;

05. Employees Without Project
select employee_id ,
first_name
from employees 
where employee_id not in (select employee_id from employees_projects)
order by employee_id desc
limit 3;

06. Employees Hired After
select 
first_name,
last_name,
hire_date,
(
	select name 
    from departments
    where department_id = e.department_id
)as dept_name
from employees as e 
having dept_name in ('Sales' , 'Finance')
order by hire_date;

07. Employees with Project
select 
e.employee_id,
e.first_name,
p.name as project_name
from employees as e
join employees_projects as ep on ep.employee_id = e.employee_id
join projects as p on p.project_id = ep.project_id
where p.start_date < '2022-08-13' and p.end_date is null
order by e.first_name,project_name
limit 5;

08. Employee 24
select 
e.employee_id,
e.first_name,
IF(Year(p.start_date) >= 2005, NULL, p.name) as project_name
from employees as e
join employees_projects as ep on ep.employee_id = e.employee_id
join projects as p on p.project_id = ep.project_id
where e.employee_id = 24
order by project_name;

09. Employee Manager
select 
employee_id,
first_name,
manager_id,
(select first_name 
	from employees 
	where employee_id = e.employee_id
) as manager_name
from employees as e
having manager_id in (3 , 7)
order by first_name;


10. Employee Summary
select 
employee_id,
concat(first_name ,' ',last_name) as employee_name,
(select concat(first_name ,' ',last_name)
	from employees 
	where employee_id = e.manager_id
) as manager_name,
(select name
	from departments 
	where department_id = e.department_id
) as manager_name
from employees as e
order by employee_id;


11. Min Average Salary
select 
avg(salary)
from employees as e
join departments as d on d.department_id = e.department_id
group by e.department_id 
order by avg(salary)
limit 1;
